def minmax_algorithm(state, depth, is_maximizing):
    if depth == 0 or game_over(state):
        return evaluate_state(state)
    
    if is_maximizing:
        max_eval = float('-inf')
        for move in get_possible_moves(state):
            eval = minmax_algorithm(
                make_move(state, move), 
                depth - 1, False)
            max_eval = max(max_eval, eval)
        return max_eval
    else:
        min_eval = float('inf')
        for move in get_possible_moves(state):
            eval = minmax_algorithm(
                make_move(state, move), 
                depth - 1, True)
            min_eval = min(min_eval, eval)
        return min_eval
